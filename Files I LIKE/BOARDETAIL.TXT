// src/pages/BoardDetail.tsx
import { useEffect, useState } from "react";
import { useParams, useNavigate } from "react-router-dom";
import { Button } from "@/components/ui/button";
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from "@/components/ui/alert-dialog";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";

import Navbar from "@/components/Navbar";
import MasonryGrid from "@/components/MasonryGrid";
import PinCard from "@/components/PinCard";
import EditBoardDialog from "@/components/EditBoardDialog";
import AddPinToBoardDialog from "@/components/AddPinToBoardDialog";

import { Plus, Trash2, ArrowLeft, Pencil, Camera } from "lucide-react";
import { useAuth } from "@/contexts/AuthContext";
import { useToast } from "@/hooks/use-toast";

interface Pin {
  _id: string;
  title?: string;
  description?: string;
  mediaUrl?: string;
  image?: string;
  mediaType?: "image" | "video";
  category?: string;
}

interface Board {
  _id: string;
  name: string;
  description?: string;
  coverImage?: string;
  user?: {
    _id: string;
    username: string;
  };
  pins: Pin[];
  createdAt?: string;
}

const BoardDetail = () => {
  const { boardId } = useParams<{ boardId: string }>();
  const navigate = useNavigate();
  const { user: currentUser } = useAuth();
  const { toast } = useToast();

  const [board, setBoard] = useState<Board | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  const [isAddDialogOpen, setIsAddDialogOpen] = useState(false);
  const [isEditDialogOpen, setIsEditDialogOpen] = useState(false);
  const [isDeleteDialogOpen, setIsDeleteDialogOpen] = useState(false);
  const [isCoverDialogOpen, setIsCoverDialogOpen] = useState(false);

  const [coverFade, setCoverFade] = useState(false);

  const apiUrl =
    import.meta.env.VITE_API_URL ||
    "https://pinterest-backend-088x.onrender.com";

  const getImage = (x?: string) => x || "";

  const loadBoard = async () => {
    if (!boardId) return;

    setIsLoading(true);
    try {
      const token = localStorage.getItem("auth_token");

      const res = await fetch(`${apiUrl}/api/boards/${boardId}`, {
        headers: { Authorization: `Bearer ${token}` },
      });

      if (!res.ok) throw new Error("Failed to load board");

      const data = await res.json();
      const b: Board = (data.board || data) as Board;

      setBoard(b);
    } catch (err) {
      toast({
        title: "Error",
        description: "Failed to load board",
        variant: "destructive",
      });
    }
    setIsLoading(false);
  };

  useEffect(() => {
    loadBoard();
  }, [boardId, isAddDialogOpen]);

  const isOwnBoard =
    currentUser &&
    board?.user &&
    board.user._id === currentUser.id;

  const handleUpdateCover = async (imageUrl: string) => {
    if (!board || !boardId) return;

    setCoverFade(true);

    setTimeout(async () => {
      try {
        const token = localStorage.getItem("auth_token");

        const res = await fetch(`${apiUrl}/api/boards/${boardId}`, {
          method: "PUT",
          headers: {
            Authorization: `Bearer ${token}`,
            "Content-Type": "application/json",
          },
          body: JSON.stringify({
            name: board.name,
            description: board.description,
            coverImage: imageUrl,
          }),
        });

        if (!res.ok) throw new Error("Failed to update cover");

        setBoard((prev) => (prev ? { ...prev, coverImage: imageUrl } : prev));

        toast({
          title: "Success",
          description: "Board cover updated",
        });
      } catch {
        toast({
          title: "Error",
          description: "Failed to update cover",
          variant: "destructive",
        });
      }

      setCoverFade(false);
      setIsCoverDialogOpen(false);
    }, 150);
  };

  if (isLoading || !board) {
    return (
      <div className="min-h-screen bg-background">
        <Navbar />
        <main className="container mx-auto px-4 py-8">
          <div className="flex justify-center items-center min-h-[300px] text-muted-foreground">
            Loading boardâ€¦
          </div>
        </main>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-background">
      <Navbar />

      <main className="container mx-auto px-4 py-8">
        <Button
          variant="ghost"
          onClick={() => navigate(-1)}
          className="gap-2 mb-4"
        >
          <ArrowLeft className="w-4 h-4" />
          Back
        </Button>

        {/* Cover */}
        <div
          className={`relative rounded-lg overflow-hidden transition-opacity duration-200 ${
            coverFade ? "opacity-0" : "opacity-100"
          }`}
        >
          {board.coverImage ? (
            <img
              src={getImage(board.coverImage)}
              className="w-full aspect-[16/9] object-cover"
            />
          ) : board.pins.length ? (
            <img
              src={getImage(board.pins[0].mediaUrl || board.pins[0].image)}
              className="w-full aspect-[16/9] object-cover"
            />
          ) : (
            <div className="w-full aspect-[16/9] bg-muted flex items-center justify-center">
              No cover image
            </div>
          )}

          {isOwnBoard && (
            <button
              onClick={() => setIsCoverDialogOpen(true)}
              className="absolute top-3 right-3 bg-white/90 rounded-full p-2 shadow"
            >
              <Camera className="w-5 h-5" />
            </button>
          )}
        </div>

        {/* Title + Buttons */}
        <div className="flex items-start justify-between mt-6">
          <div>
            <h1 className="text-4xl font-bold">{board.name}</h1>
            {board.description && (
              <p className="text-muted-foreground mt-1">
                {board.description}
              </p>
            )}
            <p className="text-sm text-muted-foreground mt-2">
              {board.pins.length} pins
            </p>
          </div>

          {isOwnBoard && (
            <div className="flex gap-2">
              <Button onClick={() => setIsAddDialogOpen(true)}>
                <Plus className="w-4 h-4 mr-2" />
                Add Pin
              </Button>

              <Button
                variant="outline"
                onClick={() => setIsEditDialogOpen(true)}
              >
                <Pencil className="w-4 h-4 mr-2" />
                Edit
              </Button>

              <AlertDialog
                open={isDeleteDialogOpen}
                onOpenChange={setIsDeleteDialogOpen}
              >
                <AlertDialogTrigger asChild>
                  <Button variant="destructive">
                    <Trash2 className="w-4 h-4 mr-2" />
                    Delete
                  </Button>
                </AlertDialogTrigger>

                <AlertDialogContent>
                  <AlertDialogHeader>
                    <AlertDialogTitle>Delete board?</AlertDialogTitle>
                    <AlertDialogDescription>
                      This cannot be undone.
                    </AlertDialogDescription>
                  </AlertDialogHeader>

                  <AlertDialogFooter>
                    <AlertDialogCancel>Cancel</AlertDialogCancel>
                    <AlertDialogAction
                      onClick={async () => {
                        try {
                          const token = localStorage.getItem("auth_token");
                          await fetch(`${apiUrl}/api/boards/${board._id}`, {
                            method: "DELETE",
                            headers: {
                              Authorization: `Bearer ${token}`,
                            },
                          });
                          navigate(-1);
                        } catch {
                          toast({
                            title: "Error",
                            description: "Failed to delete board",
                            variant: "destructive",
                          });
                        }
                      }}
                    >
                      Delete
                    </AlertDialogAction>
                  </AlertDialogFooter>
                </AlertDialogContent>
              </AlertDialog>
            </div>
          )}
        </div>

        {/* Pins */}
        {board.pins.length ? (
          <MasonryGrid>
            {board.pins.map((pin) => (
              <PinCard
                key={pin._id}
                id={pin._id}
                imageUrl={pin.mediaUrl || pin.image}
                mediaType={pin.mediaType as "image" | "video" | undefined}
                title={pin.title}
                description={pin.description}
                hideBoardButton
              />
            ))}
          </MasonryGrid>
        ) : (
          <div className="text-center py-20 text-muted-foreground">
            No pins yet.
          </div>
        )}

        {/* Cover selection dialog */}
        <Dialog open={isCoverDialogOpen} onOpenChange={setIsCoverDialogOpen}>
          <DialogContent className="max-w-xl">
            <DialogHeader>
              <DialogTitle>Select cover image</DialogTitle>
            </DialogHeader>

            <div className="grid grid-cols-3 gap-3 max-h-64 overflow-y-auto">
              {board.pins.map((pin) => {
                const img = pin.mediaUrl || pin.image;
                if (!img) return null;

                return (
                  <div
                    key={pin._id}
                    onClick={(e) => {
                      e.stopPropagation();
                      handleUpdateCover(img);
                    }}
                    className="cursor-pointer rounded-lg overflow-hidden border hover:border-primary"
                  >
                    <img
                      src={img}
                      className="w-full h-24 object-cover"
                    />
                  </div>
                );
              })}
            </div>
          </DialogContent>
        </Dialog>

        {/* Add pins to board */}
        <AddPinToBoardDialog
          open={isAddDialogOpen}
          onOpenChange={setIsAddDialogOpen}
          boardId={board._id}
          existingPins={board.pins.map((p) => p._id)}
        />

        {/* Edit board */}
        <EditBoardDialog
          open={isEditDialogOpen}
          onOpenChange={setIsEditDialogOpen}
          board={board}
          onSave={(b) => setBoard(b)}
        />
      </main>
    </div>
  );
};

export default BoardDetail;
